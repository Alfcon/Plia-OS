**Goal:** Add a dashboard settings tab showing MCP server health, with per-server disable and full-subsystem restart.

**Architecture:** Three layers — backend functions in `core/mcp_client.py` expose status/disable/restart logic; three new API endpoints in `dashboard/server.py` wrap them; a new "MCP" pane in `dashboard/static/index.html` renders the list and wires up actions. Tasks are ordered by dependency: backend → API → dashboard.

**Tech Stack:** Python (asyncio, FastAPI), vanilla JS, HTML/CSS inline styles (matching existing panel patterns).

## Global Constraints

- No new dependencies
- Follow existing inline-style patterns in `index.html` — no external CSS files
- `formatUptime(seconds)`: `< 60` → `"45s"`, `< 3600` → `"14m 32s"`, `≥ 3600` → `"2h 14m"`, `null` → `"—"`
- Status values are exactly: `"connected"`, `"disabled"`, `"failed"` (lowercase strings)
- MCP nav button goes between Modules and Home in `#m-nav`
- MCP pane goes between `#m-section-modules` and `#m-section-home`
- `asyncio_mode = "auto"` in pytest.ini — async tests need no `@pytest.mark.asyncio` decorator
- Run full suite with: `source .venv/bin/activate && pytest --tb=short -q`

```python
def get_mcp_status() -> list[dict]:
    if not _MCP_CONFIG.exists():
        return []
    try:
        servers = json.loads(_MCP_CONFIG.read_text())
    except Exception:
        return []
    now = time.monotonic()
    result = []
    for cfg in servers:
        name = cfg.get("name", "")
        if not name:
            continue
        if name in _disabled_servers:
            status = "disabled"
        elif name in _servers:
            status = "connected"
        else:
            status = "failed"
        srv = _servers.get(name)
        result.append({
            "name": name,
            "status": status,
            "tools": list(srv.tools) if srv else [],
            "uptime_seconds": (now - srv.started_at) if srv else None,
        })
    return result


def disable_mcp_server(name: str) -> bool:
    if not _MCP_CONFIG.exists():
        return False
    try:
        servers = json.loads(_MCP_CONFIG.read_text())
    except Exception:
        return False
    known = {cfg.get("name") for cfg in servers}
    if name not in known:
        return False
    _disabled_servers.add(name)
    if name in _servers:
        _servers[name].healthy = False
    return True


async def restart_mcp_servers() -> None:
    global _exit_stack, _initialized
    async with _restart_lock:
        await _exit_stack.aclose()
        _exit_stack = AsyncExitStack()
        _servers.clear()
        _disabled_servers.clear()
        _initialized = False
        await load_mcp_servers()
```

```javascript
  function formatUptime(seconds) {
    if (seconds === null || seconds === undefined) return '—';
    seconds = Math.floor(seconds);
    if (seconds < 60) return `${seconds}s`;
    if (seconds < 3600) {
      const m = Math.floor(seconds / 60);
      const s = seconds % 60;
      return `${m}m ${s}s`;
    }
    const h = Math.floor(seconds / 3600);
    const m = Math.floor((seconds % 3600) / 60);
    return `${h}h ${m}m`;
  }

  async function loadMcpServers() {
    const list = document.getElementById('mcp-servers-list');
    list.innerHTML = '<li style="color:#555;font-size:0.75rem;">Loading...</li>';
    try {
      const r = await fetch('/api/mcp/servers');
      if (!r.ok) throw new Error(r.status);
      const servers = await r.json();
      if (servers.length === 0) {
        list.innerHTML = '<li style="color:#555;font-size:0.75rem;">No MCP servers configured.</li>';
        return;
      }
      list.innerHTML = servers.map(s => {
        const dot = s.status === 'connected' ? '●' : s.status === 'disabled' ? '○' : '✕';
        const dotColor = s.status === 'connected' ? '#4caf50' : s.status === 'disabled' ? '#555' : '#e53935';
        const canDisable = s.status === 'connected';
        const btnStyle = canDisable
          ? 'background:#1e1e1e;border:1px solid #333;border-radius:3px;color:#4fc3f7;font-size:0.7rem;padding:2px 6px;cursor:pointer;flex-shrink:0;'
          : 'background:#1e1e1e;border:1px solid #222;border-radius:3px;color:#333;font-size:0.7rem;padding:2px 6px;cursor:default;flex-shrink:0;';
        const toolCount = s.tools.length;
        const uptime = formatUptime(s.uptime_seconds);
        return `<li style="padding:6px 0;border-bottom:1px solid #1a1a1a;display:flex;align-items:center;gap:8px;">
          <span style="color:${dotColor};font-size:1rem;flex-shrink:0;">${dot}</span>
          <div style="flex:1;min-width:0;">
            <strong style="color:#e0e0e0;font-size:0.78rem;">${_esc(s.name)}</strong>
            <span style="color:#555;font-size:0.7rem;margin-left:6px;">${_esc(s.status)}</span>
            <span style="color:#666;font-size:0.7rem;margin-left:6px;">${toolCount} tool${toolCount !== 1 ? 's' : ''}</span>
            <span style="color:#444;font-size:0.7rem;margin-left:6px;">${uptime}</span>
          </div>
          <button onclick="${canDisable ? `disableMcpServer('${_esc(s.name)}')` : ''}"
            style="${btnStyle}" ${canDisable ? '' : 'disabled'}>
            ${canDisable ? 'Disable' : '—'}
          </button>
        </li>`;
      }).join('');
    } catch(e) {
      list.innerHTML = '<li style="color:#ef9a9a;font-size:0.75rem;">Failed to load MCP servers.</li>';
    }
  }

  async function disableMcpServer(name) {
    try {
      const r = await fetch(`/api/mcp/servers/${encodeURIComponent(name)}/disable`, {method: 'POST'});
      if (!r.ok) throw new Error(r.status);
      await loadMcpServers();
    } catch(e) {
      alert(`Failed to disable MCP server: ${name}`);
    }
  }

  async function restartMcp() {
    const btn = document.getElementById('mcp-restart-btn');
    btn.textContent = 'Restarting…';
    btn.style.pointerEvents = 'none';
    try {
      const r = await fetch('/api/mcp/restart', {method: 'POST'});
      if (!r.ok) throw new Error(r.status);
      await new Promise(resolve => setTimeout(resolve, 1500));
      await loadMcpServers();
    } catch(e) {
      alert('MCP restart failed');
    } finally {
      btn.textContent = '↻ Restart MCP';
      btn.style.pointerEvents = '';
    }
  }
```
